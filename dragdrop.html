<button id="next-btn" onclick="generateHuruf()">Lanjut</button>


<script>

    let skor = 0;

    let hurufBesar = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";

    let hurufKecil = "abcdefghijklmnopqrstuvwxyz";

    let currentHuruf = "";


    function generateHuruf() {

        // Reset tampilan

        document.getElementById("feedback").innerHTML = "";

        document.getElementById("next-btn").style.display = "none";


        // Ambil huruf acak

        let randomIndex = Math.floor(Math.random() * 26);

        currentHuruf = hurufBesar[randomIndex];


        // Isi elemen

        document.getElementById("huruf").textContent = currentHuruf;

        document.getElementById("target").textContent = hurufKecil[randomIndex];

    }


    // Event drag

    document.getElementById("huruf").addEventListener("dragstart", function(event) {

        event.dataTransfer.setData("text", event.target.textContent);

    });


    // Event drop

    document.getElementById("target").addEventListener("dragover", function(event) {

        event.preventDefault();

    });


    document.getElementById("target").addEventListener("drop", function(event) {

        event.preventDefault();

        let draggedHuruf = event.dataTransfer.getData("text");


        if (draggedHuruf.toLowerCase() === currentHuruf.toLowerCase()) {

            skor++;

            document.getElementById("feedback").innerHTML = "✔ Benar! Hebat!";

            document.getElementById("feedback").style.color = "#2B7A78";

        } else {

            document.getElementById("feedback").innerHTML = "✖ Salah, coba lagi!";

            document.getElementById("feedback").style.color = "red";

        }


        document.getElementById("score").innerHTML = "Skor: " + skor;

        document.getElementById("next-btn").style.display = "inline-block";

    });


    generateHuruf();

</script>
